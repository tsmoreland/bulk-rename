set (SERVICE_LIBRARY_NAME ${CMAKE_PROJECT_NAME}_service)

set (SOURCE )

add_library(${SERVICE_LIBRARY_NAME} STATIC ${SOURCE})
set target_properties(${SERVICE_LIBRARY_NAME} PROPERTIES VS_GLOBAL_KEYWORD "Win32Proj")
set_target_properties(${SERVICE_LIBRARY_NAME} PROPERTIES
	OUTPUT_DIRECTORY_DEBUG   "${CMAKE_SOURCE_DIR}/target/${CMAKE_VS_PLATFORM_NAME}/$<CONFIG>/"
	OUTPUT_DIRECTORY_RELEASE "${CMAKE_SOURCE_DIR}/target/${CMAKE_VS_PLATFORM_NAME}/$<CONFIG>/"
)

target_compile_definitions(${SERVICE_LIBRARY_NAME} PRIVATE ${COMMON_COMPILE_DEFINITIONS})

if (MSVC) 
    target_compile_options(${SERVICE_LIBRARY_NAME} PRIVATE ${COMMON_COMPILE_OPTIONS})
    target_link_options(${SERVICE_LIBRARY_NAME} PRIVATE ${COMMON_LINK_OPTIONS})
endif()
